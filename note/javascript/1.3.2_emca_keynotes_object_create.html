    # 2.1构造函数模式

    1、创建一个构造函数

    function Person(name,age,job){
    this.name = name;
    this.name = age;
    this.job = job;
    this.sayName= function (){console.log(this.name)}
}
var person1 = new Person('caomei',29,"Software Enginner")

    2、实现一个new // 我们观察上面的构造函数。没有显示创建对象。 // 直接将属性和方法 给this对象。 // 没有return语句。

    //因此我们new的方法： new 是一个函数。传入 构造函数 和 参数。 创建一个新对象。新对象的proto 绑定 构造函数的原型。 执行构造函数中的代码。通过apply，传入参数。 返回 newobj。 如果构造函数有return，则返回构造函数的return。

        function Person(name,age){
        this.name = name
        this.age = age
    }
    function _new(fn,...args){
        let new_obj = Object.create(fn.prototype)//let new_obj = {}, new_obj.__proto__  = fn.prototype
        let result  = fn.apply(new_obj,args) //剩余函数就类数组。
        //没有返回对象，res不是一个对象，则返回this，也就是obj
        return typeof result ==="object" ? result : new_obj
    }
    let  p  = _new(Person,'caomei','18')
    console.log(Person.prototype == p.__proto__) //true
    console.log(p instancof Person) //true

    3.实现instanceof 涉及到原型链了。 instancof 用法是 。p instanof Person。构造函数的原型属性 是否在p的原型链子上。 1、判断p.proto 和Person的prototype是否相等。 2、如果不相等找 p.__proto__上的原型链继续网上找。 函数传入2个值，一个p，一个person。

    function _instanceof(p,Person){
    left = p.__proto__ //对象的隐式原型 
    right = Person.prototype //构造函数的显示原型。
    while (true){
        if (right === null){
            return false
        }
        if(left === right){
            return true
        }
        left = left.__proto__
    }
}
let a ='123'
console.log(_instanceof(a,Object))

    # 2.2原型模式

    1.理解 构造函数，构造函数的原型，构造函数的实例对象。

    function Person(){
}
Person.prototype.name ='caomei'
Person.prototype.sayName = function (){
    alert(this.name)
}
obj1 = new Person()//caomei
obj2 = new Person()//caomei

    //构造函数:Person()

    //构造函数的原型:Person.prototype

    //构造函数的实例：obj1

    // Person -> Person.prototype <- person.proto

    // Person < - Person.propotype.constructor -<person.proto //原型链

    1. 原型链--- 原型和实例 关系的api
    // instanceof  (判断 obj1实例的原型链子上 是否有 prototype原型)
console.log( obj1 instanceof Person) //ture
console.log( Object instanceof Object) //ture
console.log( Function  instanceof Function) //ture
console.log( String  instanceof String) // false
console.log( Number instanceof Number) //false

    //获取 obj1实例 上的 构造函数原型。 (已经知道1个，求另一个)
console.log(Object.getPrototypeOf(person1) ==Person.prototype) //ture
console.log(Object.getPrototypeOf(person1) ==Person.prototype) //caomei

    //判断 实例是否在 构造函数原型上。  (已经知道两个，求两个的关系)
console.log(Person.prototype.isPrototypeOf(obj1)) //true
